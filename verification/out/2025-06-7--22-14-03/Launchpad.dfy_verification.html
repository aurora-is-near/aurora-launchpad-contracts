datatype Option = None | Some(T)
datatype Result = Ok(T) | Err(E)
datatype Mechanics = FixedPrice(price: nat) | PriceDiscovery

class Discount {
  var startDate: nat
  var endDate: nat
  var percentage: nat

  constructor(start: nat, end: nat, perc: nat)
    requires perc <= 10000
    requires start < end
  {
    startDate := start;
    endDate := end;
    percentage := perc;
  }

  predicate IsActive(time: nat) reads this {
    startDate <= time < endDate
  }
}

class Config {
  var Discount: seq
}
